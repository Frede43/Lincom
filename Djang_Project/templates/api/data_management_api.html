Supported Formats

Available formats: CSV JSON Excel

Data Import

POST /api/data/import/

Import data from file

Request Body:
{
    "type": "startup_data",
    "format": "csv",
    "file_url": "https://example.com/data.csv",
    "mapping": {
        "columns": {
            "name": "startup_name",
            "description": "company_description",
            "founded_date": "date_founded"
        },
        "transformations": {
            "founded_date": "date_format(YYYY-MM-DD)"
        }
    },
    "options": {
        "skip_header": true,
        "batch_size": 1000,
        "duplicate_handling": "update"
    },
    "validation": {
        "rules": [
            {
                "field": "name",
                "required": true,
                "min_length": 2
            }
        ]
    }
}
Response:
{
    "job_id": "import_123",
    "status": "processing",
    "total_records": 5000,
    "processed_records": 0,
    "estimated_time": 300  // seconds
}
GET /api/data/import/{job_id}/status/

Check import job status

Data Export

POST /api/data/export/

Export data to file

Request Body:
{
    "type": "analytics_report",
    "format": "excel",
    "filters": {
        "date_range": {
            "start": "2024-01-01",
            "end": "2024-01-31"
        },
        "categories": ["tech", "agriculture"],
        "status": "active"
    },
    "columns": [
        "id",
        "name",
        "revenue",
        "employees"
    ],
    "options": {
        "include_metadata": true,
        "sheet_name": "Rapport Q1 2024",
        "password_protected": false
    }
}
GET /api/data/export/{job_id}/download/

Download exported file

Data Validation

POST /api/data/validate/

Validate data against schema

Request Body:
{
    "schema": {
        "name": {
            "type": "string",
            "required": true,
            "min_length": 2,
            "max_length": 100
        },
        "email": {
            "type": "email",
            "required": true,
            "unique": true
        },
        "founded_date": {
            "type": "date",
            "format": "YYYY-MM-DD"
        }
    },
    "data": [
        {
            "name": "TechStart Burundi",
            "email": "contact@techstart.bi",
            "founded_date": "2024-01-01"
        }
    ]
}

Data Transformation

POST /api/data/transform/

Transform data format

Request Body:
{
    "source_format": "csv",
    "target_format": "json",
    "transformations": [
        {
            "field": "date",
            "operation": "format",
            "from": "DD/MM/YYYY",
            "to": "YYYY-MM-DD"
        },
        {
            "field": "amount",
            "operation": "multiply",
            "value": 100
        }
    ],
    "options": {
        "encoding": "utf-8",
        "timezone": "Africa/Bujumbura"
    }
}

Data Backup

POST /api/data/backup/

Create data backup

Request Body:
{
    "type": "full",  // or "incremental"
    "include": [
        "startups",
        "users",
        "analytics"
    ],
    "storage": {
        "provider": "s3",
        "bucket": "backups",
        "path": "2024/01"
    },
    "encryption": {
        "enabled": true,
        "algorithm": "AES-256"
    },
    "compression": true,
    "retention": {
        "days": 30,
        "copies": 3
    }
}
GET /api/data/backup/

List data backups

Data Migration

POST /api/data/migration/

Create data migration

Request Body:
{
    "name": "update_startup_schema",
    "description": "Add new fields to startup model",
    "changes": [
        {
            "model": "Startup",
            "operation": "add_field",
            "field": {
                "name": "social_impact",
                "type": "text",
                "nullable": true
            }
        }
    ],
    "dependencies": [
        "migration_20240101"
    ],
    "rollback": {
        "enabled": true,
        "steps": [
            {
                "operation": "remove_field",
                "field": "social_impact"
            }
        ]
    }
}
POST /api/data/migration/{migration_id}/run/

Run data migration
